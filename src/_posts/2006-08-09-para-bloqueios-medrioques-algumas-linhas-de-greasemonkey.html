// ==/UserScript==(function() {
var css = document.createElement('style');
var rule = document.createTextNode('*, div, p { -moz-user-select: text !important; }');
css.appendChild(rule);
document.getElementsByTagName('body')[0].appendChild(css);
})();</strike> </i></blockquote>
